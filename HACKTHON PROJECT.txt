    for(i=1;i<=n;i++)

int hcf(int n1, int n2) {
    if (n2 != 0)
        return hcf(n2, n1 % n2);
    else
        return n1;

void removingDuplicate()//array.h
{
    int a[50],i,j,k,number;
    printf("Enter size of the array\n");
    scanf("%d",&number);
    printf("Enter Elements of the array:\n");
    for(i=0;i<number;i++)
    {
        scanf("%d",&a[i]);
    }
    printf("Entered element are: \n");
    for(i=0;i<number;i++)
    {
        printf("%d\n",a[i]);
    }
    for(i=0;i<number;i++)
    {
        for(j = i+1; j < number; j++)
            {
                if(a[i] == a[j])
                    {
                        for(k = j; k <number; k++)
                        {
                            a[k] = a[k+1];
                        }
                        j--;
                    number--;
                    }
            }
    }
    printf("\nAfter deleting the duplicate element the Array is:\n");
    for(i=0;i<number;i++)
    {
        printf("%d\n",a[i]);
    }
}
